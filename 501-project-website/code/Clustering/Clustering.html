def hier(X,kmax):
    X=np.ascontiguousarray(X) 
    params=[]; 
    scores=[]
    max_sil=-10

    for i in range(2,kmax+1):
        pred = AgglomerativeClustering(n_clusters=i).fit_predict(X)
        #pred = model.predict(X)

        try:
            scores.append(silhouette_score(X,pred))
            params.append(i)
        except:
            continue 
        
        if(scores[-1]>max_sil):
            max_sil = scores[-1]
            optimal = pred

    fig, ax = plt.subplots()
    ax.plot(params, scores, c="r")  
    ax.set_xlabel('Values of K', fontsize=14)
    ax.set_ylabel('Silhouette Score', fontsize=14)
    ax.set_title("Silhouette Score vs. K", fontsize=18)
    plt.show()

    return optimal

from scipy.cluster.hierarchy import dendrogram, linkage

# Create linkage matrix and the dendrogram using the dendrogram function in sklearn website
def dendrogram_plot(model, **kwargs):
    count = np.zeros(model.children_.shape[0])
    num_labels = len(model.labels_)
    for i, merge in enumerate(model.children_):
        current = 0
        for index in merge:
            if index < num_labels:
                current += 1
            else:
                current += count[index - num_labels]
        count[i] = current

    linkage_matrix = np.column_stack([model.children_, model.distances_, count]).astype(float)
    dendrogram(linkage_matrix, **kwargs)

plt.rcParams["figure.figsize"] = (10,10)
dendrogram_plot(model, truncate_mode='level',p=3)
